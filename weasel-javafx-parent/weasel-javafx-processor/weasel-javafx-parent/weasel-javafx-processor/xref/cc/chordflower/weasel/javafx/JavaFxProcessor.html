49  /**
50   * Processor for the view annotation.
51   */
52  @AutoService( Processor.class )
53  @Internal
54  public class JavaFxProcessor extends AbstractProcessor {
55  
56    private Messager messenger;
57  
58    private Filer filer;
59  
60    private Elements elementUtils;
61  
62    @Override
63    public Set< String > getSupportedAnnotationTypes( ) {
64      return Set.of( View.class.getCanonicalName() );
65    }
66  
67    /**
68     * The version of the generated source, that in this case is Java 17.
69     */
70    @Override
71    public SourceVersion getSupportedSourceVersion( ) {
72      return SourceVersion.RELEASE_17;
73    }
74  
75    @Override
76    public synchronized void init( ProcessingEnvironment processingEnv ) {
77      super.init( processingEnv );
78      messenger = processingEnv.getMessager();
79      filer = processingEnv.getFiler( );
80      elementUtils = processingEnv.getElementUtils( );
81    }
82  
83    /**
84     * Processes all the annotated types and then for each type:
85     *
86     * 1. Generates a new class, that inherits from the annotated type.
87     * 2. Associates the fxml indicated by the **View** annotation to the generated class.
88     * 3. Creates a new constructor that:
89     * 3.1. Calls the super constructor with any arguments it receives.
90     * 3.2. Initializes all fields annotated with Component with the corresponding field.
91     * 3.3. Binds all methods annotated with EventFor to the corresponding event.
92     * 3.4. Annotates the constructor with Inject, to allow for Dagger injection.
93     * 4. Adds the annotation Named with the name of the annotated type (in lower camel case).
94     *
95     * @param annotations The detected annotations.
96     * @param roundEnv The environment.
97     * @return Always true.
98     */
99    @Override
100   public boolean process( Set< ? extends TypeElement > annotations, @NotNull RoundEnvironment roundEnv ) {
101     for( Element element : roundEnv.getElementsAnnotatedWith( View.class ) ) {
102       if( element.getKind().equals( ElementKind.CLASS ) && !element.getModifiers().contains( Modifier.FINAL ) ) {
103         // Process element
104         this.processFile( (TypeElement) element );
105       } else {
106         this.messenger.printMessage( Kind.MANDATORY_WARNING, "The view annotation can only be placed in a non-final class", element );
107       }
108     }
109     return true;
110   }
111 
112   /**
113    * Generates the java file.
114    *
115    * @param element The element to generate the java file for.
116    */
117   @SuppressWarnings( { "OptionalGetWithoutIsPresent", "ResultOfMethodCallIgnored" } )
118   private void processFile( TypeElement element ) {
119 
120     var information = TypeInformation.getTypeInformation( element, false );
121 
122     // 1. Get the fxml filename...
123     var fxmlFilename = information.Annotation( View.class ).get().name();
124 
125     // 2. Find out the name of the class
126     var className = information.Name();
127     if( className.isEmpty( ) ) {
128       this.messenger.printMessage( Kind.MANDATORY_WARNING, "Unable to get the class name, make sure the annotated class is not an anonymous class!", element );
129       return;
130     }
131 
132     // 3. Create a new type named <CLASS_NAME>Robodog
133     var typeSpec = TypeSpec.classBuilder(String.format("%sWeasel", className))
134         .addJavadoc( "This type contains the implementation of the view class without the View annotation\n" )
135         .addJavadoc( "as generated by the weasel-javafx-processor" )
136         .addModifiers( Modifier.PUBLIC, Modifier.FINAL );
137 
138     // 4. Add the Named annotation to the new class
139     typeSpec.addAnnotation( AnnotationSpec.builder( Generated.class )
140         .addMember( "value", "$S", "weasel-javafx-processor" )
141         .addMember( "date", "$S", LocalDate.now().format( DateTimeFormatter.ISO_DATE ) )
142         .addMember( "comments", "$S", "Can't touch this!" )
143         .build() );
144     typeSpec.addAnnotation( AnnotationSpec.builder( Named.class )
145         .addMember( "value", "$S", CaseFormat.UPPER_CAMEL.to( CaseFormat.LOWER_CAMEL, className ) )
146         .build() );
147 
148     // 5. The new type should extend the class
149     typeSpec.superclass( element.asType() );
150 
151     // 6. Get the arguments of the class constructor
152     var constructor = information.Constructors().stream()
153         .filter( el -> !el.Modifiers().contains( TypeModifiersEnum.PRIVATE ) ).findFirst();
154     if( constructor.isEmpty() ) {
155       this.messenger.printMessage( Kind.MANDATORY_WARNING, "The given class does not have a valid constructor!", element );
156       return;
157     }
158 
159     // 7. Create a new constructor that has the same arguments as the class constructor
160     var newConstructor = MethodSpec.constructorBuilder().addModifiers( Modifier.PUBLIC );
161 
162     for( var parameter : constructor.get( ).Parameters() ) {
163       var typeName = TypeName.get( parameter.ParameterType() );
164       newConstructor.addParameter( typeName, parameter.Name() );
165     }
166 
167     // 8. Add the Injected annotation to the constructor
168     newConstructor.addAnnotation( Inject.class );
169 
170     // 9. Make a call to the super constructor in the new class
171     var res = String.join( ",", constructor.get().Parameters().stream().map( ParameterInformation::Name ).toList() );
172     newConstructor.addStatement( "super($L)", res );
173 
174     /* 10. Add the code for loading the fxml file to the constructor:
175      *
176      * FXMLLoader fxmlLoader = new FXMLLoader(getClass().getResource("<FXML_FILENAME>"));
177      * fxmlLoader.setRoot(this);
178      * fxmlLoader.setController(this);
179      *
180      * try {
181      *   fxmlLoader.load();
182      * } catch( IOException exception ) {
183      *   throw new RuntimeException(exception);
184      * }
185      */
186     var fxmlLoader = ClassName.get( "javafx.fxml", "FXMLLoader" );
187     newConstructor.addComment( "Attempt to load the given fxml file, inside the root provided by this view" );
188     newConstructor.addStatement( "$T fxmlLoader = new $T( getClass().getResource( $S ) )", fxmlLoader, fxmlLoader, fxmlFilename );
189     newConstructor.addStatement( "fxmlLoader.setRoot( this )" );
190     newConstructor.addStatement( "fxmlLoader.setController( this )" );
191     newConstructor.addCode( "" );
192     newConstructor.beginControlFlow( "try" );
193     newConstructor.addStatement( "fxmlLoader.load()" );
194     newConstructor.nextControlFlow( "catch( $T ex )", IOException.class );
195     newConstructor.addStatement("throw new $T(ex)", RuntimeException.class);
196     newConstructor.endControlFlow();
197     newConstructor.addCode( "\n" );
198 
199     if( information.Fields().stream().anyMatch( el -> el.hasAnnotation( Component.class ) && !el.Modifiers().contains( TypeModifiersEnum.PRIVATE ) ) ) {
200       newConstructor.addComment( "Inject the fields from the fxml" );
201     }
202 
203     // 11. For each field annotated with Component (and not private):
204     for( var field : information.Fields().stream().filter( el -> el.hasAnnotation( Component.class ) && !el.Modifiers().contains( TypeModifiersEnum.PRIVATE ) ).toList() ) {
205       // 11.1. Get the id from the annotation
206       var id = String.format( "#%s", field.Annotation( Component.class ).get().id() );
207 
208       // 11.2. Get the name of the field
209       var name = field.Name();
210 
211       // 11.3. Get the type of the field
212       var fieldType = (( MirroredFieldInformation<?> )field).Field().asType();
213 
214       /* 11.4. Add the following code to the constructor:
215        *
216        * this.<FIELD_NAME> = JavaFxUtils.getInstance().<<CONTROL_TYPE>>lookup( "#<ID>", this ).get();
217        */
218       newConstructor.addStatement( "this.$L = $T.getInstance().<$T>lookup( $S, this ).get()", name, ClassName.get( JavaFxUtils.class ), fieldType, id );
219     }
220     newConstructor.addCode( "\n" );
221 
222     if( information.Methods().stream().anyMatch( el -> ( el.hasAnnotation( EventFor.class ) || el.hasAnnotation( EventsFor.class ) ) && !el.Modifiers().contains( TypeModifiersEnum.PRIVATE ) ) ) {
223       newConstructor.addComment( "Inject the events in the proper event handler" );
224     }
225 
226     // 12. For each method annotated with EventFor ( and not private ):
227     for( var method : information.Methods().stream().filter( el -> el.hasAnnotation( EventFor.class ) && !el.Modifiers().contains( TypeModifiersEnum.PRIVATE ) ).toList() ) {
228       // 12.1. Get the id from the annotation
229       var id = String.format( "#%s", method.Annotation( EventFor.class ).get().id() );
230 
231       // 12.2. Get the action from the annotation
232       var action = String.format( "%sProperty", method.Annotation( EventFor.class ).get().action() );
233 
234       // 12.3. Get the control from the annotation
235       TypeMirror control = null;
236       try {
237         method.Annotation( EventFor.class ).get().control( );
238       } catch( MirroredTypeException ex ) {
239         control = ex.getTypeMirror();
240       }
241 
242       // 12.4. Get the name of the method
243       var methodName = method.Name();
244 
245       /* 12.5 Add the following code to the constructor:
246        *
247        * (JavaFxUtils.getInstance().<<CONTROL_TYPE>>lookup( "<ID>" ).get()).<ACTION>Property().set( this::<METHOD_NAME> );
248        */
249       newConstructor.addStatement( "($T.getInstance().<$T>lookup( $S, this ).get()).$L().set( this::$L )", ClassName.get( JavaFxUtils.class ), control, id, action, methodName );
250     }
251 
252     // 13. For each method annotation with EventsFor ( and not private ):
253     for( var method : information.Methods().stream().filter( el -> el.hasAnnotation( EventsFor.class ) && !el.Modifiers().contains( TypeModifiersEnum.PRIVATE ) ).toList() ) {
254 
255       // 13.1. Get the EventFor list form the annotation
256       var eventsFor = method.Annotation( EventsFor.class ).get().value();
257 
258       // 13.2. For each EventFor
259       for( EventFor eventFor : eventsFor ) {
260 
261         // 13.2.1. Get the id from the annotation
262         var id = String.format( "#%s", eventFor.id() );
263 
264         // 13.2.2. Get the action from the annotation
265         var action = String.format( "%sProperty", eventFor.action() );
266 
267         // 13.2.3. Get the control from the annotation
268         TypeMirror control = null;
269         try {
270           eventFor.control( );
271         } catch( MirroredTypeException ex ) {
272           control = ex.getTypeMirror();
273         }
274 
275         // 13.2.4. Get the name of the method
276         var methodName = method.Name();
277 
278         /* 13.2.5 Add the following code to the constructor:
279          *
280          * (JavaFxUtils.getInstance().<<CONTROL_TYPE>>lookup( "<ID>" ).get()).<ACTION>Property().set( this::<METHOD_NAME> );
281          */
282         newConstructor.addStatement( "($T.getInstance().<$T>lookup( $S, this ).get()).$L().set( this::$L )", ClassName.get( JavaFxUtils.class ), control, id, action, methodName );
283       }
284     }
285 
286     // 14. If there is an method called init without any arguments call it
287     if( information.Methods().stream().anyMatch( el -> "init".equals( el.Name() ) &&
288         !el.Modifiers().contains( TypeModifiersEnum.PRIVATE ) && el.Parameters().isEmpty() ) ) {
289       newConstructor.addCode( "\n" );
290       newConstructor.addComment( "Call the init method" );
291       newConstructor.addStatement( "this.init()" );
292     }
293 
294     // 15. Generate the class and place it in the outdir
295     var result = typeSpec.addMethod( newConstructor.build() ).build();
296 
297     var javaFile = JavaFile.builder( this.elementUtils.getPackageOf( element ).getQualifiedName().toString(), result )
298         .build();
299 
300     try {
301       javaFile.writeTo( this.filer );
302     } catch( IOException e ) {
303       this.messenger.printMessage( Kind.ERROR, e.getMessage() );
304     }
305   }
