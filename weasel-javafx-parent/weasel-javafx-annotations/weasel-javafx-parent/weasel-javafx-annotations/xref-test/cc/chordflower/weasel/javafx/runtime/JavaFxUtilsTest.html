66      @Test
67      @DisplayName( "it should be able to search for an id in it" )
68      public void test01( ) {
69  
70        List< Triplet< String, Node, ? > > provideArguments = this.provideArguments( );
71        for( var i = 0; i < provideArguments.size( ); i++ ) {
72          Triplet< String, Node, ? > arg = provideArguments.get( i );
73          System.out.printf( "Running test %d\n", i );
74          var result = JavaFxUtils.getInstance( ).lookup( arg.getValue0( ), arg.getValue1( ) );
75          Matchers.expect( result.isPresent( ) ).toBeTrue( );
76          Matchers.expect( result.orElse( null ) ).toBeInstanceOf( arg.getValue2( ).getClass( ) );
77          Matchers.expect( result.orElse( null ) ).toEqual( arg.getValue2( ) );
78        }
79      }
